Complete Router Configuration Script

If you need to configure router manually or auto-push fails, use this complete configuration:

SSH to router and paste this:

# MikroTik Router Configuration - Team 11
# SSH to router: ssh admin@172.18.13.11
# Password: CantGuessThis67!

# Disable insecure services
/ip service disable telnet,ftp,www,api,api-ssl,winbox
/ip service set ssh port=22

# Configure interfaces
/interface ethernet
set [ find default-name=ether1 ] comment="WAN/External" name=ether1
set [ find default-name=ether2 ] comment="LAN/Internal" name=ether2

# Set IP addresses
/ip address
add address=172.18.13.11/16 interface=ether1 comment="ncae-wan"
add address=192.168.11.1/24 interface=ether2 comment="ncae-lan"

# NAT masquerade (critical for internal VMs to reach internet)
/ip firewall nat
add chain=srcnat out-interface=ether1 action=masquerade comment="ncae-masquerade"

# Port forwards for scoring
add chain=dstnat dst-address=172.18.13.11 protocol=tcp dst-port=80 action=dst-nat to-addresses=192.168.11.5 to-ports=80 comment="ncae-http"
add chain=dstnat dst-address=172.18.13.11 protocol=tcp dst-port=443 action=dst-nat to-addresses=192.168.11.5 to-ports=443 comment="ncae-https"
add chain=dstnat dst-address=172.18.13.11 protocol=tcp dst-port=53 action=dst-nat to-addresses=192.168.11.12 to-ports=53 comment="ncae-dns-tcp"
add chain=dstnat dst-address=172.18.13.11 protocol=udp dst-port=53 action=dst-nat to-addresses=192.168.11.12 to-ports=53 comment="ncae-dns-udp"

# Firewall rules
/ip firewall filter
# Allow established/related
add chain=input connection-state=established,related action=accept comment="ncae-input-established"
add chain=forward connection-state=established,related action=accept comment="ncae-forward-established"

# Allow SSH to router
add chain=input protocol=tcp dst-port=22 action=accept comment="ncae-ssh"

# Allow ICMP (scoring!)
add chain=input protocol=icmp action=accept comment="ncae-icmp"

# Allow from internal LAN
add chain=input src-address=192.168.11.0/24 action=accept comment="ncae-lan-input"

# Forward internal to external
add chain=forward src-address=192.168.11.0/24 action=accept comment="ncae-lan-forward"

# Forward port forwards to internal
add chain=forward dst-address=192.168.11.5 protocol=tcp dst-port=80,443 action=accept comment="ncae-web-forward"
add chain=forward dst-address=192.168.11.12 protocol=tcp dst-port=53 action=accept comment="ncae-dns-tcp-forward"
add chain=forward dst-address=192.168.11.12 protocol=udp dst-port=53 action=accept comment="ncae-dns-udp-forward"

# Drop everything else
add chain=input action=drop comment="ncae-input-drop"
add chain=forward action=drop comment="ncae-forward-drop"

# Set default route (if needed)
/ip route
add dst-address=0.0.0.0/0 gateway=172.18.0.1 comment="ncae-default-route"

# Change admin password (do this LAST!)
/user set admin password=NewSecurePassword123!
